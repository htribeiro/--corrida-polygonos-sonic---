Como fazer o commit e push (passo a passo — rodar no seu terminal)

  1. Se ainda não tem o arquivo local, crie-o:
  • Abra um editor e salve o conteúdo acima como index.html dentro de uma pasta nova, por exemplo corrida-polygonos.
  1. Inicializar git, commitar e enviar para o repositório remoto (HTTPS):

  • Abra o terminal na pasta onde está index.html e execute: git init git adicionar index.html git commit -m "Adicionar index.html para a Corrida dos Polígonos com Sonic" git branch -M main git remote add origin https://github.com/htribeiro/--corrida-polygonos-sonic---.git git push -u origin main

Observação: se o repositório no GitHub já existe e está vazio, esses passos criarão a branch main e enviarão o commit. Se houver arquivos remotos na branch main, antes puxe ou faça um clone e faça um merge/resolução de conflitos.

<h1>Corrida dos Polígonos com Sonic 🌀</h1>

<button class="botao" onclick="jogar()">🎲 Jogar</button>
<div class="resultado" id="resultado"></div>

<script>
const casas = [
{ forma: "🔺 Triângulo", desafio: "Conte os lados e vértices", pontos: 2 },
{ forma: "⬛ Quadrado", desafio: "Desenhe um escudo", pontos: 3 },
{ forma: "🟦 Retângulo", desafio: "Mostre um objeto retangular", pontos: 2 },
{ forma: "🔷 Pentágono", desafio: "Pinte a forma", pontos: 2 },
{ forma: "🟩 Hexágono", desafio: "Crie um padrão com hexágonos", pontos: 3 },
{ forma: "🟧 Octógono", desafio: "Diga o poder da forma", pontos: 2 },
{ forma: "🌟 Curinga", desafio: "Imite o Sonic!", pontos: 2 }
];

let posicao = 0;
let pontuacao = 0;

function desenharTabuleiro() {
const tabuleiro = document.getElementById("tabuleiro");
casas.forEach((casa, i) => {
const div = document.createElement("div");
div.className = "casa";
div.innerHTML = `<strong>${casa.forma}</strong><br><small>${casa.desafio}</small>`;
tabuleiro.appendChild(div);
});
}

function jogar() {
const dado = Math.floor(Math.random() * 6) + 1;
posicao = (posicao + dado) % casas.length;
const casa = casas[posicao];
pontuacao += casa.pontos;
document.getElementById("resultado").innerHTML =
`🎲 Você rolou ${dado} e caiu na casa ${posicao + 1} (${casa.forma})<br>
Desafio: ${casa.desafio}<br>
Pontuação: ${pontuacao}`;
}

desenharTabuleiro();
</script>
</body>
</html>
